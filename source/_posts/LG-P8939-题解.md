STL 题。

首先观察题目中给的排列的权值，利用交换法容易发现按照从小到大的顺序排列最优。而整个的权值就是差分数组的和（规定差分数组第一项为 $0$，下同）加上 $nums_{n} - nums_{1}$，我们又知道整个差分数组的和就是 $nums_{n} - nums_{1}$，因此最终的权值就是 $2 \times (mx - mn)$，$mx,mn$ 分别表示序列中最大的数和最小的数。

利用 `multiset` 维护这个序列，由于相同大小的元素是等价的，可以直接判断 $x$ 与 `*st.begin()` 和 `*st.rbegin()` 来判断 $x$ 是否是最大、最小的数。

时间复杂度：$O(n \log n)$。

```cpp
#include <bits/stdc++.h>
using namespace std;
#define int long long
#define max_n 1010101
const long double eps = 1e-15;
void read(int &p)
{
    p = 0;
    int k = 1;
    char c = getchar();
    while (c < '0' || c > '9')
    {
        if (c == '-')
        {
            k = -1;
        }
        c = getchar();
    }
    while (c >= '0' && c <= '9')
    {
        p = p * 10 + c - '0';
        c = getchar();
    }
    p *= k;
    return;
}
void write_(int x)
{
    if (x < 0)
    {
        putchar('-');
        x = -x;
    }
    if (x > 9)
    {
        write_(x / 10);
    }
    putchar(x % 10 + '0');
}
void writesp(int x)
{
    write_(x);
    putchar(' ');
}
void writeln(int x)
{
    write_(x);
    putchar('\n');
}
int n, m;
int nums[max_n];
multiset<int> st;
int sum = 0;
signed main()
{
    read(n), read(m);
    for (int i = 1; i <= n; i++)
    {
        read(nums[i]);
        st.insert(nums[i]);
    }
    sort(nums + 1, nums + n + 1);
    sum = 2 * (nums[n] - nums[1]);
    for (int i = 1, op, x; i <= m; i++)
    {
        read(op), read(x);
        // 发现如果不是最前、最后没有影响

        if (op == 1)
        {
            auto it = st.find(x);
            if (it == st.end())
            {
                puts("-1");
                continue;
            }
            // 在开头
            if (it == st.begin())
            {
                sum -= 2 * abs(*st.rbegin() - *st.begin());
                st.erase(it);
                sum += 2 * abs(*st.rbegin() - *st.begin());
            }
            else if (*it == *st.rbegin())
            {
                sum -= 2 * abs(*st.rbegin() - *st.begin());
                st.erase(it);
                sum += 2 * abs(*st.rbegin() - *st.begin());
            }
            else
            {
                st.erase(it);
            }
        }
        else
        {
            // 在开头
            if (x < *st.begin())
            {
                sum -= 2 * abs(*st.rbegin() - *st.begin());
                st.insert(x);
                sum += 2 * abs(*st.rbegin() - *st.begin());
            }
            else if (x > *st.rbegin())
            {
                sum -= 2 * abs(*st.rbegin() - *st.begin());
                st.insert(x);
                sum += 2 * abs(*st.rbegin() - *st.begin());
            }
            else
            {
                st.insert(x);
            }
        }
        writeln(sum);
    }
    return 0;
}
```
