有两个很容易发现的性质：

1. 如果两次操作的区间有重叠，一定是不优的，原因是这样至少会多 $k$ 的花费，并且对实现序列全部相同的效果不如直接选择一个大区间。

2. 最终所有数一定都等于原数组所有数的 $\gcd$，不断取 $\gcd$ 使得最终所有数全相同一定会全部成为 $\gcd$。

因此，我们最终的操作一定是在数组中截取若干个不相交的区间进行操作。设 $m$ 为整个序列的 $\gcd$。

选择整个序列操作一次一定能把所有数变成相同的，但这样花费不一定是最优的。考虑什么时候我们会把一个区间分成两段：发现如果中间存在一段长度为 $l$（$l > k$）的数均为 $m$，这段区间是不需要选择的，这样可以减少 $l - k$ 的花费（中间一段不用选，减少 $l$，多一次操作增加 $k$）。不过还需要考虑两侧区间的 $\gcd$ 是否为 $m$，如果不是还要带一个中间的 $m$，这样我们能达成更优的 $l$ 就会增加 $1$ 或 $2$。

实现的时候可以先假设每一段 $m$ 都不选，然后再合并。


```cpp
#include <bits/stdc++.h>
using namespace std;
#define int long long
#define max_n 4111110
void read(int &p)
{
    p = 0;
    int k = 1;
    char c = getchar();
    while (c < '0' || c > '9')
    {
        if (c == '-')
        {
            k = -1;
        }
        c = getchar();
    }
    while (c >= '0' && c <= '9')
    {
        p = p * 10 + c - '0';
        c = getchar();
    }
    p *= k;
    return;
}
void write_(int x)
{
    if (x < 0)
    {
        putchar('-');
        x = -x;
    }
    if (x > 9)
    {
        write_(x / 10);
    }
    putchar(x % 10 + '0');
}
void writesp(int x)
{
    write_(x);
    putchar(' ');
}
void writeln(int x)
{
    write_(x);
    putchar('\n');
}
int n, k, tot, flag;
int nums[max_n];
int ans;
pair<int, int> seg[max_n >> 1];
signed main()
{
    read(n), read(k);
    for (int i = 1, ls = 1; i <= n; i++)
    {
        read(nums[i]);
    }
    flag = nums[1];
    for (int i = 2; i <= n; i++)
    {
        if (flag != 1)
        {
            flag = __gcd(flag, nums[i]);
        }
        else
        {
            break;
        }
    }
    int ls = 1;
    for (int i = 1; i <= n; i++)
    {
        if (nums[i] == flag)
        {
            if (i - ls > 0)
            {
                seg[++tot] = {ls, i - 1};
            }
            ls = i + 1;
        }
    }
    if (ls <= n)
    {
        seg[++tot] = {ls, n};
    }
    if (tot == 0)
    {
        puts("0");
        return 0;
    }
    ls = 1;
    for (int i = 2; i <= tot; i++)
    {
        // 长度太小，分成两段不优
        if (seg[i].first - 1 - seg[i - 1].second <= k)
        {
            continue;
        }
        if (i - ls > 1)
        {
            ans += k + (seg[i - 1].second - seg[ls].first + 1);
            ls = i;
        }
        else
        {
            int _ = nums[seg[i - 1].first];
            for (int j = seg[i - 1].first + 1; j <= seg[i - 1].second; j++)
            {
                _ = __gcd(_, nums[j]);
                if (_ == flag)
                {
                    break;
                }
            }
            // 这一段不需要占用一个
            if (_ == flag)
            {
                ans += k + (seg[i - 1].second - seg[i - 1].first + 1);
                ls = i;
            }
            else if (seg[i].first - 1 - seg[i - 1].second > k + 1)
            {
                ans += k + (seg[i - 1].second - seg[i - 1].first + 2);
                ls = i;
            }
        }
    }
    int i = tot + 1;
    if (i - ls > 1)
    {
        ans += k + (seg[i - 1].second - seg[ls].first + 1);
        ls = i;
    }
    else
    {
        int _ = nums[seg[i - 1].first];
        for (int j = seg[i - 1].first + 1; j <= seg[i - 1].second; j++)
        {
            _ = __gcd(_, nums[j]);
            if (_ == flag)
            {
                break;
            }
        }
        // 这一段不需要占用一个
        if (_ == flag)
        {
            ans += k + (seg[i - 1].second - seg[i - 1].first + 1);
        }
        else
        {
            ans += k + (seg[i - 1].second - seg[i - 1].first + 2);
        }
        ls = i;
    }
    writeln(ans);
    return 0;
}
```
